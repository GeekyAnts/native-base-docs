export default class DrawerExample extends Component {
    render() {
      closeDrawer = () => {

        this.drawer._root.open()
      };
        return (
            &lt;Drawer
              ref={(ref) => { this.drawer = ref; }}
              content={&lt;SideBar navigator={this.navigator} />}
              onClose={() => this.closeDrawer()} >
            // Main View
          &lt;/Drawer>
        );
    }
